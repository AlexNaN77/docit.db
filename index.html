// Room Entity ejemplo (Kotlin)
@Entity(tableName = "doctors")
data class Doctor(
    @PrimaryKey val id: String,
    val name: String,
    val specialty: String,    // "cardiologia", "odontologia"...
    val photoUrl: String?,    // base64 o URL
    val phone: String?,
    val address: String,
    val mapsUrl: String?,
    val schedule: String?,
    val services: String,     // JSON serializado de [{name, price}]
    val notes: String?,
    val uploadedBy: String,
    val createdAt: Long
)
